def named_tags
  named_tags = log.named_tags.reject{ | k, _ | TAGS_TO_REMOVE.include?(k) }
  return if named_tags.nil? || named_tags.empty?

  list = []
  named_tags.each_pair { |name, value| list << "#{name}: #{value}" }
  "{#{list.join(', ')}}"
end
