def create(uuid, environment)
  environment.uuid = uuid
  if environment.display_name.blank?
    environment.display_name = PactBroker::Pacticipants::GenerateDisplayName.call(environment.name)
  end
  environment.save
end

def replace(uuid, environment)
  environment.uuid = uuid
  if environment.display_name.blank?
    environment.display_name = PactBroker::Pacticipants::GenerateDisplayName.call(environment.name)
  end
  environment.upsert
end
