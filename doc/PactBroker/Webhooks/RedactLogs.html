def self.call logs, values
  substitutions = HEADER_SUBSTITUTIONS + value_substitutions(values)

  substitutions.reduce(logs) do | agg_logs, (find, replace) |
    agg_logs.gsub(find, replace)
  end
end

def self.value_substitutions(values)
