def access_allowed?(env, level)
  !!rules.find do | rule_level, allowed_methods, path_pattern |
    level_allowed?(level, rule_level) &&
      method_allowed?(env, allowed_methods) &&
      path_allowed?(env, path_pattern)
  end
end
